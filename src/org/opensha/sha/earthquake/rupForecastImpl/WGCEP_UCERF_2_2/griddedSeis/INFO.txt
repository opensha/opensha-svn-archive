This is what I did to get ascii rate files here

	ls ../*/binaryAgrid/* ../*/mmax/* > junk1
	awk '{print "java Bin2Ascii4Floats",$1}' <junk1> shscr1
	chmod a+x shscr1
	shscr1
	mkdir asc_agrid
	mv ../*/binaryAgrid/*asc ../*/mmax/*asc asc_agrid

The "LonsLats.txt" file comes from an ascii file "agrid_cstcal.txt" that came from Golden at some point.

"RELM_RegionLatsLons.txt" I made from org.opensha.data.region.EvenlyGriddedRELM_Region

	ls asc_agrid/* > junk2
	awk '{print "paste", $1," >> all_data"}' <junk2> shscr2
	rm all_data
	chmod a+x shscr2
	shscr2

	paste LonsLats.txt asc_agrid/agrd_brawly.out.asc asc_agrid/agrd_creeps.out.asc asc_agrid/agrd_cstcal.out.asc asc_agrid/agrd_deepn.out.asc asc_agrid/agrd_deeps.out.asc > junk3
	paste asc_agrid/agrd_mendos.out.asc asc_agrid/agrd_wuscmp.out.asc asc_agrid/agrd_wuscmp_nopuget.out.asc asc_agrid/agrd_wusext.out.asc asc_agrid/area1new.agrid.asc > junk4
